#!/usr/bin/python3
"""Defines a text-indentation function."""

def text_indentation(text):
    """Print text with two new lines after each '.', '?', and ':'.
    Args:
        text (string): The text to print.
    Raises:
        TypeError: If text is not a string.
    """
    if not isinstance(text, str):
        raise TypeError("text must be a string")
    
    skip_space = False  # To handle consecutive spaces after '.', '?', ':'

    for char in text:
        if char in ".?:":  # If the character is a delimiter
            print(char, end="\n\n")  # Print it with two new lines
            skip_space = True
        elif char == " " and skip_space:  # Skip consecutive spaces
            continue
        else:
            print(char, end="")  # Print other characters as is
            skip_space = False
